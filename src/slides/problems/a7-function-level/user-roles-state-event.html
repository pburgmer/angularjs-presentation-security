module.run(function($rootScope, UserService, growl) {
  $rootScope.$on('$stateChangeStart', function (event, toState) {
    if (toState.userRoleRequired) {
      UserService.hasRoles(toState.userRoleRequired).then(function () {
        // to late to cancel event
        event.preventDefault();
      });
    }
  });
});
